      ### Fetch Wallets
      First, let's fetch the wallets information of the user using the [`getWallets`](/sdk-reference/react-native/getWallets) method. We will continue developing in the `user-profile.tsx`.

      ```tsx filename="screens/user-profile.tsx" copy
      import { useOkto, type OktoContextType, type Wallet } from 'okto-sdk-react-native';
      import React, { useState } from 'react';
      import { View, Text } from "react-native";

      const UserProfileScreen = () => {
          const { getWallets } = useOkto() as OktoContextType;
          const [userWallets, setUserWallets] = useState<Wallet[]>([]);

          React.useEffect(() => {
              getWallets()
                    .then((result) => {
                      setUserWallets(result);
                    })
                    .catch((error) => {
                      console.error(`error:`, error);
                    });
          }, []);

          return (
              <View style={{ flex: 1, backgroundColor: '#fff' }}>
                  <Text>User Wallets</Text>
                  <View>
                      {userWallets.map((wallet, index) => (
                          <View key={index} style={{ marginVertical: 10 }}>
                              <Text>{wallet.network_name}</Text>
                              <Text>{wallet.address}</Text>
                          </View>
                      ))}
                  </View>
              </View>
          );
      };

      export default UserProfileScreen;
      ```

      ### Fetch Wallets
      First, let's fetch the wallets information of the user using the [`getWallets`](/sdk-reference/flutter/getWallets) method. We will continue developing in the `user_profile.dart`.

      ```dart filename="lib/screens/user_profile.dart" copy
      import 'package:flutter/material.dart';
      import 'package:okto_sdk_flutter/okto_sdk_flutter.dart';

      class UserProfileScreen extends StatefulWidget {
        @override
        _UserProfileScreenState createState() => _UserProfileScreenState();
      }

      class _UserProfileScreenState extends State<UserProfileScreen> {
        List userWallets = [];

        @override
        void initState() {
          super.initState();
          _fetchWallets();
        }

        Future<void> _fetchWallets() async {
        List wallets = await okto.getWallets();
        setState(() {
            userWallets = wallets;
        });
        }

        @override
        Widget build(BuildContext context) {
          return Scaffold(
            appBar: AppBar(
              title: Text('User Wallets'),
            ),
            body: Center(
              child: userWallets.isEmpty
                  ? CircularProgressIndicator()
                  : ListView.builder(
                      itemCount: userWallets.length,
                      itemBuilder: (context, index) {
                        return ListTile(
                          title: Text(userWallets[index].networkName),
                          subtitle: Text(userWallets[index].address),
                        );
                      },
                    ),
            ),
          );
        }
      }
      ```

          return Scaffold(
            appBar: AppBar(
              title: Text('User Wallets'),

              child: ListView.builder(

                    title: Text(userWallets[index].networkName),
                    subtitle: Text(userWallets[index].address),
                  );
                },
              ),
            ),
          );
        }
      }
