        ```bash copy
        curl -X GET '${baseUrl}/api/v1/portfolio'
        -H 'x-api-key: YOUR_CLIENT_API_KEY'
        -H 'accept: application/json' 
        -H 'Authorization: Bearer AUTH_TOKEN'
        ```
