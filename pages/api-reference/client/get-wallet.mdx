Fetch all the wallets created by the user.

<ColumnLayout>
    <>
        ## Headers
        `Authorization` : string <br />
        Authorization token received from Okto.

        ---
        
        ## Parameters
        None
    </>
    <>
        ## Code
        ```bash copy
        curl -X GET '${baseUrl}/api/v1/wallet' 
        -H 'x-api-key: YOUR_CLIENT_API_KEY'
        -H 'accept: application/json' 
        -H 'Authorization: Bearer AUTH_TOKEN'
        ```

        ## Response
        ```json
        {
            "status": "success",
            "data": {
                "wallets": [
                    {
                        "network_name": "POLYGON_TESTNET",
                        "address": "0x0342A54DD44E8744FD185579Af57845Cb0ac6cB0",
                        "success": true
                    }
                ]
            }
        }
        ```
    </>
</ColumnLayout>
